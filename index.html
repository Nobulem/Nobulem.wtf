// Load latest games from games page async function loadLatestGames() { try { const response = await fetch('/games/index.html'); const html = await response.text(); const parser = new DOMParser(); const doc = parser.parseFromString(html, 'text/html'); const gameCards = doc.querySelectorAll('.game-card'); const games = []; gameCards.forEach((card, index) => { if (index < 3) { const thumbnail = card.querySelector('.game-thumbnail'); const name = card.querySelector('.game-name'); const description = card.querySelector('.game-description'); const gameId = card.querySelector('.game-id'); const statusBadge = card.querySelector('.status-badge'); if (thumbnail && name && description) { games.push({ name: name.textContent.trim(), description: description.textContent.trim(), image: thumbnail.src || thumbnail.getAttribute('src'), id: gameId ? gameId.textContent.trim().replace('#', '').trim() : '', status: statusBadge ? statusBadge.textContent.trim() : 'Working' }); } } }); if (games.length > 0) { renderLatestGames(games); } else { renderLatestGames(getFallbackGames()); } } catch (error) { console.error('Could not fetch games:', error); renderLatestGames(getFallbackGames()); } }
